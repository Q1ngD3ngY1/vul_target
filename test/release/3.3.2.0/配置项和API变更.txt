------------2.新增定时任务--未完成的文档解析定时轮询任务 begin------------
trpc_go.yaml
server:
  service:
    - name: trpc.adp.kb_config.UnfinishedDocParseRefresh #service的路由名称,
      nic: eth1 #网卡 ipport配置二选一，主要用于分布式定时器的互斥
      port: 9105 #服务监听端口 可使用占位符 ${port}
      # 定时器cron表达式: [second minute hour day month weekday]
      network: "0 */5 * * * *?scheduler=UnfinishedDocParseRefresh&holdTime=5"
      protocol: timer #应用层协议
      timeout: 900000 # 请求最长处理时间，单位毫秒

main.yaml
# 未完成文档解析刷新配置
unfinished_doc_parse_refresh_config:
  enable: true
  parse_status_store_time_second: 604800 # 7天
  refresh_batch_size: 50
  ignore_file_parse_callback_for_debug: false #线上绝对不能开！线上绝对不能开！线上绝对不能开！测试环境开启后，会忽略文件解析回调，用于测试未完成文档解析刷新配置的定时任务
